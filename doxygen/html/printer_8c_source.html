235int64_t PrintNSBUFVA(int stream, Buffer_t* buffer, int64_t used, const char* format, va_list va) {
236 int64_t bufferUsed = used;

242 .format = format,
243 .current = current,
244 .bufferUsed = &bufferUsed,
245 };

247 if (NULL == format) {
248 goto ERROR_EXIT;
249 }
250

252 switch (*current) {

259 __Percentage(&data, va);

264 buffer->str[bufferUsed++] = *current;
265 goto SKIP_TYPE_SWITCH;
266 }
267SKIP_TYPE_SWITCH:
268 {}
269 }
270END_LOOP:

272ERROR_EXIT:
